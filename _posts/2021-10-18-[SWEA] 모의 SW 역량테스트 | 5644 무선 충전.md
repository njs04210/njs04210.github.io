from collections import deque

dx, dy = [0,0,1,0,-1],[0,-1,0,1,0]
T = int(input())
for test_case in range(1, T + 1):
    board = [[list() for _ in range(11)] for _ in range(11)]

    m, bc = map(int, input().split()) # ì´ë™ì‹œê°„, BCê°œìˆ˜
    user_A = [0]+list(map(int, input().split()))
    user_B = [0]+list(map(int, input().split()))
    bc_power = []
    answer = 0

    for i in range(bc):
        x, y, c, p = map(int,input().split())
        bc_power.append(p)
        ch = set()
        q = deque()

        q.append((y,x))
        ch.add((y,x))
        while q:
            tmp_y, tmp_x = q.popleft()
            board[tmp_y][tmp_x].append(i)
            for j in range(1, 5):
                ny = tmp_y + dx[j]
                nx = tmp_x + dy[j]
                if nx < 1 or nx > 10 or ny < 1 or ny > 10: continue
                if (ny,nx) in ch: continue
                if abs(nx-x) + abs(ny-y) <= c:
                    q.append((ny,nx))
                    ch.add((ny,nx))

    a, b = [1,1], [10,10]
    for i in range(m+1):
        dir_a, dir_b = user_A[i], user_B[i]
        a[0] += dy[dir_a]
        a[1] += dx[dir_a]
        b[0] += dy[dir_b]
        b[1] += dx[dir_b]

        a_candi= board[a[0]][a[1]]
        b_candi = board[b[0]][b[1]]

        if not a_candi and not b_candi: continue

        Maxi = -2147000000
        if not a_candi:
            for bv in b_candi:
                if bc_power[bv] > Maxi:
                    Maxi = bc_power[bv]
        elif not b_candi:
            for av in a_candi:
                if bc_power[av] > Maxi:
                    Maxi = bc_power[av]
        else:
            for av in a_candi:
                for bv in b_candi:
                    result = 0
                    if av == bv:
                        result = bc_power[av]
                    else:
                        result += bc_power[av]
                        result += bc_power[bv]

                    if result > Maxi:
                        Maxi = result

        answer += Maxi

    print(f'#{test_case} {answer}')
