def check(a):

    cnt = 1
    for i in range(1, n):
        if a[i] == a[i-1]: # 평지
            cnt += 1
        elif a[i] - a[i-1] == 1 and cnt >= x:  # 오르막길
            cnt = 1
        elif a[i] - a[i-1] == -1 and cnt >= 0: # 내리막길
            cnt = -x+1
        else:
            return 0
    if cnt >= 0:
        return 1
    return 0

T = int(input())
for test_case in range(1, T + 1):
    n, x = map(int, input().split())
    MAP = [list(map(int, input().split())) for _ in range(n)]
    answer = 0

    for i in range(n):
        answer += check(MAP[i])

    MAP = list(zip(*MAP))
    for i in range(n):
        answer += check(MAP[i])

    print(f'#{test_case} {answer}')
