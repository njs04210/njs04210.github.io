dx, dy = ['dummy', -1, -1, 0, 1, 1, 1, 0, -1], ['dummy', 0, -1, -1, -1, 0, 1, 1, 1]
dx2, dy2 = [-1, 1, 1, -1], [-1, -1, 1, 1]
n, m = map(int, input().split())

                a[i][j] -= 2

    answer += sum(a[i])
