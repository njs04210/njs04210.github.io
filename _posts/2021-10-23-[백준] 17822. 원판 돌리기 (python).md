from collections import deque

def rotate(d, k, board):
    for _ in range(k):
        if d == 0: # ì‹œê³„
            tmp = board.pop()
            board.appendleft(tmp)
        elif d == 1:
            tmp = board.popleft()
            board.append(tmp)
    return board

dx, dy = [1, 0, -1, 0], [0, 1, 0, -1]
n, m, t = map(int, input().split())
board = []
for _ in range(n):
    board.append(deque(list(map(int, input().split()))))

for _ in range(t):
    x, d, k = map(int, input().split())

    # 1. íšŒì „í•˜ê¸°
    for i in range(n):
        if (i+1) % x == 0:
            board[i] = rotate(d, k, board[i])

    # 2. ì¸ì ‘í•˜ë©´ì„œ ê°™ì€ ìˆ«ìžë©´ ì§€ìš°ê¸°
    flag2 = False
    for i in range(n):
        if sum(board[i]) == 0: continue
        for j in range(m):
            if board[i][j] == 0: continue
            q = deque()
            q.append((i, j))
            num = board[i][j]
            board[i][j] = 0
            flag = False
            while q:
                x, y = q.popleft()
                for dir in range(4):
                    nx = x + dy[dir]
                    ny = y + dx[dir]
                    if nx >= n or nx < 0: continue
                    if board[nx][ny % m] == num:
                        q.append((nx, ny % m))
                        board[nx][ny % m] = 0
                        flag = True

            if not flag:
                board[i][j] = num
                continue
            flag2 = True # í‰ê· ë‚´ê¸° ê³¼ì • ìƒëžµ ê°€ëŠ¥

    # 3. í‰ê· ë‚´ê¸°
    if not flag2:
        stack = []
        for i in range(n):
            for j in range(m):
                if board[i][j] == 0: continue
                stack.append(board[i][j])

        if not stack: break
        avg = sum(stack) / len(stack)
        for i in range(n):
            for j in range(m):
                if board[i][j] == 0: continue
                if board[i][j] < avg:
                    board[i][j] += 1
                elif board[i][j] > avg:
                    board[i][j] -= 1

answer = 0
for i in range(n):
    answer += sum(board[i])

print(answer)
