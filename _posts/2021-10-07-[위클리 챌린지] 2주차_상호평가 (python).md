```python
def getGrade(score):
    if score >= 90:
        return "A"
    elif 80 <= score < 90:
        return "B"
    elif 70 <= score < 80:
        return "C"
    elif 50 <= score < 70:
        return "D"
    else:
        return "F"

def solution(scores):
    
    p = len(scores) # 총 인원
    answer = ""
    
    scores = list(zip(*scores))
    
    for i in range(p):
        s = scores[i]
        myScore = s[i]
        mini, maxi, total = min(s), max(s), sum(s)
        
        if myScore in (mini, maxi):
            if s.count(myScore) == 1:
                answer += getGrade((total - myScore) / (p-1))
                continue
        
        answer += getGrade(total / p)
        
    return answer
```
