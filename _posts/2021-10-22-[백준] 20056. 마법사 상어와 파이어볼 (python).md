```python
from collections import deque

dx, dy = [0, 1, 1, 1, 0, -1, -1, -1], [-1, -1, 0, 1, 1, 1, 0, -1]
n, m, k = map(int, input().split())
board = [[deque() for _ in range(n)] for _ in range(n)]
q = deque()

for _ in range(m):
    x, y, m, s, d = map(int, input().split())
    board[x-1][y-1].append((m, s, d))
    q.append((x, y))

for _ in range(k):

    tmp = []
    qlen = len(q)
    for _ in range(qlen):
        x, y = q.popleft()
        for _ in range(len(board[x-1][y-1])):
            m, s, d = board[x-1][y-1].popleft()
            nx = (x-1 + dy[d] * s) % n + 1
            ny = (y-1 + dx[d] * s) % n + 1
            q.append((nx, ny))
            tmp.append([nx, ny, m, s, d])

    for x, y, m, s, d in tmp:
        board[x-1][y-1].append((m, s, d))

    for i in range(n):
        for j in range(n):
            if len(board[i][j]) >= 2:
                p = len(board[i][j])
                total_m, total_s, flag = 0, 0, 0
                for ii in range(p):
                    m, s, d = board[i][j][ii]
                    total_m += m
                    total_s += s
                    if d % 2 == 0: # ì§ìˆ˜ì¼ë•Œ
                        flag += 1
                    else: # í™€ìˆ˜ì¼ë•Œ
                        flag -= 1
                m = total_m // 5
                s = total_s // p
                board[i][j] = deque()
                if m != 0:
                    if abs(flag) == p:
                        for idx in range(4):
                            board[i][j].append((m, s, idx * 2))
                    else:
                        for idx in range(4):
                            board[i][j].append((m, s, idx * 2 + 1))

total = 0
for i in range(n):
  for j in range(n):
    if len(board[i][j]) > 0:
      for m, s, d in board[i][j]:
        total += m

print(total)
```
