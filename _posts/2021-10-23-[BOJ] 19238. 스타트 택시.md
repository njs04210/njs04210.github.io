from collections import deque

dx, dy = [1, 0, -1, 0], [0, 1, 0, -1]
n, m, oil = map(int, input().split())
MAP = [[0] * (n+1)]

for _ in range(n):
    MAP.append([0] + list(map(int, input().split())))

tx, ty = map(int, input().split())

client, end = deque(), deque()
for _ in range(m):
    ax, ay, bx, by = map(int, input().split())
    client.append((ax, ay))
    end.append((bx, by))

ch = [False] * m

for _ in range(m):
    Mini = 2147000000
    turn = -1
    next = [2147000000, 2147000000]

    # ê±°ë¦¬êµ¬í•˜ê¸°
    q = deque()
    visited = [[0] * (n+1) for _ in range(n+1)]
    visited[tx][ty] = 1
    q.append((tx, ty))

    while q:
        x, y = q.popleft()
        for i in range(4):
            nx = x + dy[i]
            ny = y + dx[i]
            if nx < 1 or nx > n or ny < 1 or ny > n: continue
            if MAP[nx][ny] == 1: continue
            if visited[nx][ny]: continue
            visited[nx][ny] = visited[x][y] + 1
            q.append((nx, ny))

    # ì†ë‹˜ ì„ íƒí•˜ê¸°
    for i in range(m):
        if ch[i]: continue
        x, y = client[i]
        dist = visited[x][y] - 1
        if 0 <= dist < Mini:
            Mini = dist
            next[0], next[1] = x, y
            turn = i
        elif dist == Mini:
            if x < next[0]:
                next[0], next[1] = x, y
                turn = i
            elif x == next[0]:
                if y < next[1]:
                    next[0], next[1] = x, y
                    turn = i
    if Mini == 2147000000:
        break
    else:
        if oil - Mini < 0: break
        oil -= Mini
        tx, ty = next[0], next[1] # íƒì‹œ ì¢Œ ê°±ì‹ 
        dest_x, dest_y = end[turn] # ëª©ì ì§€ ì¢Œí‘œ

    # ëª©ì ì§€ê¹Œì§€ ì´ë™
    q = deque()
    q.append((tx, ty, 0))
    visited = [[0] * (n+1) for _ in range(n+1)]
    visited[tx][ty] = 1
    flag = False
    while q:
        x, y, used_oil = q.popleft()

        if used_oil > oil:
            flag = True
            break
        if x == dest_x and y == dest_y: # ëª©ì ì§€ ë„ì°©í–ˆì„ë•Œ
            oil = (oil - used_oil) + (used_oil * 2)
            ch[turn] = 1
            tx, ty = dest_x, dest_y
            break
        for i in range(4):
            nx = x + dy[i]
            ny = y + dx[i]
            if nx < 1 or nx > n or ny < 1 or ny > n: continue
            if visited[nx][ny]: continue
            if MAP[nx][ny] == 1: continue
            q.append((nx, ny, used_oil + 1))
            visited[nx][ny] = 1

    if flag:
        break

if all(ch):
    print(oil)
else:
    print(-1)
