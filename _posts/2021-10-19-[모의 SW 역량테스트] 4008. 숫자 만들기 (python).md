def dfs(L, total):
    if L == n:
        global Maxi, Mini

        Maxi = max(total, Maxi)
        Mini = min(total, Mini)

        return

    if op[0]:
        op[0] -= 1
        dfs(L+1, total + a[L])
        op[0] += 1

    if op[1]:
        op[1] -= 1
        dfs(L+1, total - a[L])
        op[1] += 1

    if op[2]:
        op[2] -= 1
        dfs(L+1, total * a[L])
        op[2] += 1

    if op[3]:
        op[3] -= 1
        dfs(L+1, int(total / a[L]))
        op[3] += 1

T = int(input())
for test_case in range(1, T + 1):
    n = int(input())
    op = list(map(int, input().split()))
    a = list(map(int, input().split()))

    Maxi = -2147000000
    Mini = 2147000000
    dfs(1, a[0])
    print(f'#{test_case} {Maxi-Mini}')
