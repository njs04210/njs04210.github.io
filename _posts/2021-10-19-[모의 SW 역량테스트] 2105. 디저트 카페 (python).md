```python
dx, dy = [1,-1,-1,1], [1,1,-1,-1]
T = int(input())
for test_case in range(1, T + 1):
    n = int(input())
    board = [list(map(int, input().split())) for _ in range(n)]
    answer = -2147000000

    case = []
    for i in range(1, n-1):
        for j in range(1, n-1):
            case.append((i,j))

    for i in range(0, n-1):
        for j in range(1, n-1):
            for a, b in case:
                route = set()
                x, y = i, j
                flag = False
                route.add(board[i][j])

                # 오른쪽 아래 대각선
                for _ in range(a):
                    x += dy[0]
                    y += dx[0]
                    if x < 0 or x >= n or y < 0 or y >= n:
                        flag = True
                        break
                    if board[x][y] in route:
                        flag = True
                        break
                    route.add(board[x][y])

                if flag: continue

                # 왼쪽 아래 대각선
                for _ in range(b):
                    x += dy[1]
                    y += dx[1]
                    if x < 0 or x >= n or y < 0 or y >= n:
                        flag = True
                        break
                    if board[x][y] in route:
                        flag = True
                        break
                    route.add(board[x][y])

                if flag: continue

                # 왼쪽 위 대각선
                for _ in range(a):
                    x += dy[2]
                    y += dx[2]
                    if x < 0 or x >= n or y < 0 or y >= n:
                        flag = True
                        break
                    if board[x][y] in route:
                        flag = True
                        break
                    route.add(board[x][y])

                if flag: continue

                # 오른쪽 위 대각선
                for _ in range(b):
                    x += dy[3]
                    y += dx[3]
                    if x < 0 or x >= n or y < 0 or y >= n:
                        flag = True
                        break

                    if x == i and y == j: break
                    if board[x][y] in route:
                        flag = True
                        break
                    route.add(board[x][y])

                if flag: continue
                answer = max(answer, 2*(a+b))

    if answer == -2147000000:
        answer = -1

    print(f'#{test_case} {answer}')
```
