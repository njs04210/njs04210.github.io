```python
from collections import deque

dx, dy = [1, 0, -1, 0], [0, 1, 0, -1]
N, L, R = map(int, input().split())
A = [list(map(int, input().split())) for _ in range(N)]
answer = -1
q = deque()

while True:
    answer += 1
    isFlag = False

    visited = [[0] * N for _ in range(N)]
    for i in range(N):
        for j in range(N):
            if not visited[i][j]:
                stack = []
                q.append((i, j))
                stack.append((i,j))
                visited[i][j] = 1
                while q:
                    x, y = q.popleft()
                    for p in range(4):
                        nx = x + dy[p]
                        ny = y + dx[p]
                        if nx < 0 or nx >= N or ny < 0 or ny >= N: continue
                        if visited[nx][ny]: continue
                        gap = abs(A[nx][ny] - A[x][y])
                        if L <= gap <= R:
                            q.append((nx, ny))
                            visited[nx][ny] = 1
                            stack.append((nx, ny))
                if len(stack) > 1:
                    isFlag = True
                    new_population = sum([A[x][y] for x, y in stack]) // len(stack)
                    for x, y in stack:
                        A[x][y] = new_population
    if not isFlag:
        break

print(answer)
```
