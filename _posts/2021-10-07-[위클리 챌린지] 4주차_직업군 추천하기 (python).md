```python
def solution(table, languages, preference):
    
    table.sort()

    scores, job = [], []
    
    for t in table:
        a = t.split()
        job.append(a[0])
        tmp = a[1:]
        result = 0
        
        for i, lang in enumerate(languages):
            if lang not in tmp:
                result += 0
                continue
            result += preference[i] * (5 - tmp.index(lang))
            
        scores.append(result)
    
    return job[scores.index(max(scores))]
```
