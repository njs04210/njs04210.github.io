def del_beads(sx, sy):
    global ans1, ans2, ans3

    x, y = sx, sy - 1
    idx, cnt, move, turn, flag = 0, 0, 1, 1, False
    q = []
    while True:
        if not q:
            q.append((x,y))
        nx = x + dy2[idx]
        ny = y + dx2[idx]
        if board[nx][ny] == board[x][y]:
            q.append((nx, ny))
            if nx == 0 and ny == 0:
                if len(q) >= 4:
                    flag = True
                    for i, j in q:
                        if board[i][j] == 1: ans1 += 1
                        elif board[i][j] == 2: ans2 += 1
                        elif board[i][j] == 3: ans3 += 1
                        board[i][j] = 0

        elif board[nx][ny] != board[x][y]:
            if len(q) >= 4:
                flag = True
                for i, j in q:
                    if board[i][j] == 1: ans1 += 1
                    elif board[i][j] == 2: ans2 += 1
                    elif board[i][j] == 3: ans3 += 1
                    board[i][j] = 0
            q = []

        x, y = nx, ny
        if (x == 0 and y == 0) or board[x][y] == 0:
            return flag

        idx, cnt, move, turn = change_dir(idx, cnt, move, turn)


def change_dir(idx, cnt, move, turn):
    cnt += 1
    if cnt == turn:
        idx = (idx + 1) % 4
        cnt = 0
        move += 1
        if move % 2 == 0:
            turn += 1
            move = 0

    return idx, cnt, move, turn

def move_beads(sx, sy):
    global board

    x, y = sx, sy-1
    idx, cnt, move, turn = 0, 0, 1, 1
    beads = [board[x][y]] if board[x][y] != 0 else []
    while True:
        nx = x + dy2[idx]
        ny = y + dx2[idx]
        if board[nx][ny] != 0:
            beads.append(board[nx][ny])
        x, y = nx, ny
        if x == 0 and y == 0: break
        idx, cnt, move, turn = change_dir(idx, cnt, move, turn)

    x, y = sx, sy-1
    idx, cnt, move, turn = 0, 0, 1, 1
    board = [[0] * n for _ in range(n)]
    if beads:
        board[x][y] = beads[0]
        for b in beads[1:]:
            nx = x + dy2[idx]
            ny = y + dx2[idx]
            board[nx][ny] = b

            x, y = nx, ny
            idx, cnt, move, turn = change_dir(idx, cnt, move, turn)



dx, dy = [0, 0, -1, 1], [-1, 1, 0, 0]
dx2, dy2 = [0, 1, 0, -1], [1, 0, -1, 0]
n, m = map(int, input().split())
board = [list(map(int, input().split())) for _ in range(n)]
sx, sy = n // 2, n // 2
ans1, ans2, ans3 = 0, 0, 0

for _ in range(m):
    d, s = map(int, input().split())
    d -= 1

    # 마법 부리기
    for k in range(1, s+1):
        nx = sx + k * dy[d]
        ny = sy + k * dx[d]
        if nx < 0 or nx >= n or ny < 0 or ny >= n: break
        board[nx][ny] = 0

    # 이동 및 폭발 반복
    while True:
        move_beads(sx, sy)
        flag = del_beads(sx, sy)
        if not flag: break # 폭발할게 없을때 반복문 종료

    # 그룹찾기
    x, y = sx, sy - 1
    idx, cnt, move, turn = 0, 0, 1, 1
    flag, tmp = False, []
    while True:
        if not flag:
            beads_num, beads_cnt = board[x][y], 1
            flag = True
        nx = x + dy2[idx]
        ny = y + dx2[idx]
        if board[nx][ny] == board[x][y]:
            beads_cnt += 1
            if nx == 0 and ny == 0:
                tmp.append(beads_cnt)
                tmp.append(beads_num)
        else:
            tmp.append(beads_cnt)
            tmp.append(beads_num)
            flag = False

        x, y = nx, ny
        if (x == 0 and y == 0) or board[x][y] == 0: break
        idx, cnt, move, turn = change_dir(idx, cnt, move, turn)

    x, y = sx, sy -1
    idx, cnt, move, turn = 0, 0, 1, 1
    board = [[0] * n for _ in range(n)]

    if tmp:
        board[x][y] = tmp[0]
        for k in tmp[1:]:
            nx = x + dy2[idx]
            ny = y + dx2[idx]
            board[nx][ny] = k

            x, y = nx, ny
            if x == 0 and y == 0: break
            idx, cnt, move, turn = change_dir(idx, cnt, move, turn)

print(ans1 + 2 * ans2 + 3 * ans3)
