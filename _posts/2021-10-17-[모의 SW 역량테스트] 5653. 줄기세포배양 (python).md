dx, dy = [1,0,-1,0], [0,1,0,-1]
T = int(input())
for test_case in range(1, T + 1):
    n, m, k = map(int, input().split())
    a = [list(map(int, input().split())) for _ in range(n)]
    MAP = [[0 for _ in range(m+k)] for _ in range(n+k)]
    cell_list = [[] for _ in range(11)]

    for i in range(n):
        for j in range(m):
            if a[i][j] == 0: continue
            MAP[i+k//2][j+k//2] = a[i][j]
            cell_list[a[i][j]].append([i+k//2, j+k//2, a[i][j]])

    for time in range(1, k+1):
        for life in range(10, 0, -1):
            cells = cell_list[life]
            new, old = [], []
            for i in range(len(cells)-1, -1, -1):
                cells[i][2] -= 1
                x, y, HP = cells[i]
                if HP == -1: # í™œì„±ì‹œì‘, ë²ˆì‹
                    for d in range(4):
                        nx = x + dy[d]
                        ny = y + dx[d]
                        if MAP[nx][ny] == 0:
                            MAP[nx][ny] = life
                            new.append([nx, ny, life])
                if HP == -life:
                    old.append(i)

            for idx in old:
                cells.pop(idx)
            cells += new

    result = 0
    for i in range(1, 11):
        result += len(cell_list[i])

    print(f'#{test_case} {result}')
